# Use a base image, in this case, an official Nginx image
FROM nginx:latest

# Set the working directory to where your web application files will be copied
WORKDIR /usr/share/nginx/html

# Copy the application files from your build context (local directory) to the container
COPY . /usr/share/nginx/html

# Expose port 80 for HTTP traffic
EXPOSE 80

# Define the command to start the Nginx web server
CMD ["nginx", "-g", "daemon off;"]
